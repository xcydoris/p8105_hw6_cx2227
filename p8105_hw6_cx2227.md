``` r
birthweight = birthweight %>% 
mutate(
  babysex = factor(babysex),
    frace = factor(frace),
    mrace = factor(mrace),
    malform = factor(malform)
  )

options(scipen=999)
fit = lm(bwt ~ blength + delwt + gaweeks + malform + menarche + momage + pnumlbw + ppbmi + ppwt + wtgain,
         data = birthweight)
fit %>% broom::tidy()
```
